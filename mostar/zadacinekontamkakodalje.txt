#include <iostream>
using namespace std;
/*naæi  susjedne èlanove najmanjeg (najveæeg) èlana niza*/
int main()
{
	int br;
	cout << "Unesite broj clanova niza--->";
	cin >> br;
	int iNiz[100];
	//Unos clanova niza
	for (int i = 0; i<br; i++)
	{
		cout << i << " clan niza--->";
		cin >> iNiz[i];
	}
	//Ispis unesenog niza
	cout << "Uneseni niz: " << endl;
	for (int i = 0; i<br; i++)
	{
		cout << iNiz[i] << " ";
	}
	int nmax = 0, imax = 0;
	for (int i = 0; i<br; i++)
	{
		if (iNiz[i]>nmax)
		{
			nmax = iNiz[i];
			imax = i;
		}
	}
	if (imax == 0)
	{
		cout << "Najveci clan niza je: " << iNiz[0] << endl;
		
	}
	
	system("PAUSE");
	return 0;
}
